DJ's WebGL Tutorial 007--骨骼动画

WebGL

骨骼动画是目前最常用的模型动画，
这一节将增加对骨骼动画的支持，
使用javascript进行基于cpu的骨骼更新，
使用vertex shader进行基于gpu的顶点蒙皮。
所用模型和动画数据都导出自Unity3D。

  3.     uniform mat4 u_bones[48];\ 
  4.     attribute vec4 a_position;\ 
  5.     attribute vec4 a_weights;\ 
  6.     attribute vec4 a_indices;\ 
  7.     attribute vec2 a_uv;\ 
  8.     varying vec2 v_uv;\ 
  9.     void main()\ 
  10.     {\ 
  11.         vec4 skinned_pos_world = vec4(0.0);\

  13.         indices[0] = int(a_indices.x);\

  15.         indices[2] = int(a_indices.z);\

  20.         weights[2] = a_weights.z;\

  22.         for(int i=0; i<4; i++) {\

骨骼更新

在每帧渲染之前，先更新所有的骨骼矩阵var bones = mesh.updateBones([0, -1, 0], [0, 180, 0], [1, 1, 1])。

  1. function update_bone(mesh) { 
  2.     var clip = mesh.clips['Idle']; 
  3.  
  4.     if(mesh.time == 0) { 
  5.         mesh.time = new Date().getTime(); 
  6.     } 
  7.     var now = new Date().getTime(); 
  8.     var t = (now - mesh.time) / 1000; 
  9.     var ct = t / clip['len']; 
  10.     ct = (ct - Math.floor(ct)) * clip['len']; 
  11.  
  12.     var curves = clip['curves']; 
  13.     for(var i=0; i<curves.length; i++) { 
  14.         var c = curves[i]; 
  15.  
  16.         var path = c['path']; 
  17.         var paths = path.split('/'); 
  18.         var property = c['property']; 
  19.         var keys = c['keys']; 
  20.  
  21.         var bone = null; 
  22.         for(var j=0; j<paths.length; j++) { 
  23.             if(bone == null) { 
  24.                 bone = mesh.bone_map[paths[j]]; 
  25.             } else { 
  26.                 bone = bone['children'][paths[j]]; 
  27.             } 
  28.         } 
  29.  
  30.         var value = curve_evaluate(keys, ct); 
  31.  
  32.         switch(property) { 
  33.             case "m_LocalPosition.x": 
  34.                 bone['pos_loc'][0] = value; 
  35.                 break; 
  36.             case "m_LocalPosition.y": 
  37.                 bone['pos_loc'][1] = value; 
  38.                 break; 
  39.             case "m_LocalPosition.z": 
  40.                 bone['pos_loc'][2] = value; 
  41.                 break; 
  42.  
  43.             case "m_LocalRotation.x": 
  44.                 bone['rot_loc'].array[0] = value; 
  45.                 break; 
  46.             case "m_LocalRotation.y": 
  47.                 bone['rot_loc'].array[1] = value; 
  48.                 break; 
  49.             case "m_LocalRotation.z": 
  50.                 bone['rot_loc'].array[2] = value; 
  51.                 break; 
  52.             case "m_LocalRotation.w": 
  53.                 bone['rot_loc'].array[3] = value; 
  54.                 break; 
  55.  
  56.             case "m_LocalScale.x": 
  57.                 bone['sca_loc'][0] = value; 
  58.                 break; 
  59.             case "m_LocalScale.y": 
  60.                 bone['sca_loc'][1] = value; 
  61.                 break; 
  62.             case "m_LocalScale.z": 
  63.                 bone['sca_loc'][2] = value; 
  64.                 break; 
  65.         } 
  66.     } 
  67. }

  69. SkinnedMesh.prototype.updateBones = function(t, r, s) { 
  70.     update_bone(this); 
  71.  
  72.     var trs = Matrix4x4.TRS(t, r, s); 
  73.     var rq = Quaternion.Euler(r); 
  74.  
  75.     for(var i=0; i<this.bone_array.length; i++) { 
  76.         var bone = this.bone_array[i]; 
  77.  
  78.         if(bone.parent == null) { 
  79.             bone.pos = trs.multiplyPoint3x4(bone.pos_loc); 
  80.             bone.rot = rq.multiply(bone.rot_loc); 
  81.             bone.sca = [s[0] * bone.sca_loc[0], s[1] * bone.sca_loc[1], s[2] * bone.sca_loc[2]]; 
  82.         } else { 
  83.             bone.pos = bone.parent.mat_local_to_world.multiplyPoint3x4(bone.pos_loc); 
  84.             bone.rot = bone.parent.rot.multiply(bone.rot_loc); 
  85.  
  86.             var sca = bone.parent.sca; 
  87.             bone.sca = [sca[0] * bone.sca_loc[0], sca[1] * bone.sca_loc[1], sca[2] * bone.sca_loc[2]]; 
  88.         } 
  89.  
  90.         bone.mat_local_to_world = Matrix4x4.TQS(bone.pos, bone.rot.array, bone.sca); 
  91.     } 
  92.  
  93.     for(var i=0; i<this.bind_poses.length; i++) { 
  94.         this.bone_mats[i] = this.bone_refs[i].mat_local_to_world; 
  95.     } 
  96.  
  97.     return this.bone_mats; 
  98. };
