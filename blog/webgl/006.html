DJ's WebGL Tutorial 006--模型加载、显示

WebGL

本节将加载并显示一个3D模型，
从渲染方法上讲，和上一节的立方体没有本质区别，只是顶点数据不一样。
立方体是我们自己定义顶点数据，
而3D模型，是从资源文件读取顶点数据，
所以这一节的重点在于数据读取。

从指定url加载文件二进制数据

  1. DataLoader.LoadAsync = function(url, callback, obj) { 
  2.     var req = new XMLHttpRequest(); 
  3.     req.open("GET", url, true); 
  4.     req.responseType = 'arraybuffer'; 
  5.     req.onreadystatechange = function() { 
  6.         if(req.readyState == 4) { 
  7.             if(req.status == 200 || req.status == 0) { 
  8.                 callback(new DataView(req.response), obj); 
  9.             } else { 
  10.                 alert('Could not load ' + url + ' (' + req.status + ')'); 
  11.             } 
  12.         } 
  13.     }; 
  14.  
  15.     req.send(null); 
  16. };

  1. Mesh.Load = function(url, onload) {

读取mesh数据

  1. function on_mesh_data_load(data, mesh) { 
  2.     var offset = [0]; 
  3.     read_string(data, offset); 
  4.  
  5.     var mesh_count = DataLoader.ReadInt32(data, offset); 
  6.  
  7.     for(var i=0; i<mesh_count; i++) { 
  8.         var name = read_string(data, offset); 
  9.         var position = read_vector3(data, offset); 
  10.         var rotation = read_quaternion(data, offset); 
  11.         var scale = read_vector3(data, offset); 
  12.  
  13.         var vertex_count = DataLoader.ReadInt32(data, offset); 
  14.         var vertices = new Float32Array(vertex_count * 3); 
  15.         for(var j=0; j<vertex_count; j++) { 
  16.             var v = read_vector3(data, offset); 
  17.             vertices.set(v, j*3); 
  18.         } 
  19.         mesh.vertices = vertices; 
  20.  
  21.         var color_count = DataLoader.ReadInt32(data, offset); 
  22.         var colors = new Float32Array(color_count * 4); 
  23.         for(var j=0; j<color_count; j++) { 
  24.             //read color 
  25.             var v = read_quaternion(data, offset); 
  26.             colors.set(v, j*4); 
  27.         } 
  28.  
  29.         var uv_count = DataLoader.ReadInt32(data, offset); 
  30.         var uvs = new Float32Array(uv_count * 2); 
  31.         for(var j=0; j<uv_count; j++) { 
  32.             var x = DataLoader.ReadFloat32(data, offset); 
  33.             var y = 1 - DataLoader.ReadFloat32(data, offset); 
  34.             uvs.set([x, y], j*2); 
  35.         } 
  36.         mesh.uv = uvs; 
  37.  
  38.         var uv2_count = DataLoader.ReadInt32(data, offset); 
  39.         var uv2s = new Float32Array(uv2_count * 2); 
  40.         for(var j=0; j<uv2_count; j++) { 
  41.             var x = DataLoader.ReadFloat32(data, offset); 
  42.             var y = 1 - DataLoader.ReadFloat32(data, offset); 
  43.             uv2s.set([x, y], j*2); 
  44.         } 
  45.  
  46.         var normal_count = DataLoader.ReadInt32(data, offset); 
  47.         var normals = new Float32Array(normal_count * 3); 
  48.         for(var j=0; j<normal_count; j++) { 
  49.             var v = read_vector3(data, offset); 
  50.             normals.set(v, j*3); 
  51.         } 
  52.  
  53.         var tangent_count = DataLoader.ReadInt32(data, offset); 
  54.         var tangents = new Float32Array(tangent_count * 4); 
  55.         for(var j=0; j<tangent_count; j++) { 
  56.             //read vector4 
  57.             var v = read_quaternion(data, offset); 
  58.             tangents.set(v, j*4); 
  59.         } 
  60.  
  61.         var index_count = DataLoader.ReadInt32(data, offset); 
  62.         var indices = new Uint16Array(index_count); 
  63.         for(var j=0; j<index_count; j++) { 
  64.             var v = DataLoader.ReadUint16(data, offset); 
  65.             indices[j] = v; 
  66.         } 
  67.         mesh.indices = indices; 
  68.  
  69.         var submeshs = []; 
  70.         var submesh_count = DataLoader.ReadInt32(data, offset); 
  71.         for(var j=0; j<submesh_count; j++) { 
  72.             var sub_count = DataLoader.ReadInt32(data, offset); 
  73.             var sub_indices = new Uint16Array(index_count); 
  74.             for(var k=0; k<sub_count; k++) { 
  75.                 var v = DataLoader.ReadUint16(data, offset); 
  76.                 sub_indices[k] = v; 
  77.             } 
  78.  
  79.             submeshs[j] = sub_indices; 
  80.         } 
  81.  
  82.         break; 
  83.     } 
  84.  
  85.     mesh.loaded = true; 
  86.  
  87.     mesh.onload(mesh); 
  88. }; 

创建顶点buffer

  1. function create_buffer() { 
  2.     mesh = Mesh.Load("ranger.mesh", function(m) { 
  3.         vertex_buffer = gl.createBuffer(); 
  4.         gl.bindBuffer(gl.ARRAY_BUFFER, vertex_buffer); 
  5.         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(m.vertices), gl.STATIC_DRAW); 
  6.  
  7.         uv_buffer = gl.createBuffer(); 
  8.         gl.bindBuffer(gl.ARRAY_BUFFER, uv_buffer); 
  9.         gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(m.uv), gl.STATIC_DRAW); 
  10.  
  11.         index_buffer = gl.createBuffer(); 
  12.         gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, index_buffer); 
  13.         gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(m.indices), gl.STATIC_DRAW); 
  14.     }); 
  15. } 

初始化：

  1. function init_gl() { 
  2.     gl.clearColor(0.0, 0.0, 1.0, 1.0); 
  3.     gl.enable(gl.DEPTH_TEST); 
  4.     gl.enable(gl.CULL_FACE); 
  5.     gl.frontFace(gl.CW); 
  6.  
  7.     create_shader(); 
  8.     create_buffer(); 
  9.     create_texture(); 
  10. } 

这里新加了gl.frontFace(gl.CW)修改三角形正面的顶点顺序为CW顺时针。
因为mesh数据是我从Unity3D导出的，Unity3D引擎使用的顶点顺序是CW，
而gl默认是使用CCW逆时针。
